void putPrice(String values) {
    byte len = values.length();
    if (len >= 2 && values.substring(len - 2).equals("00")) {
        for (byte i = 0; i < len - 2; i++) {
            char value = values.charAt(i);
            if (value == '.' || value == ',') continue; 
            int getValue = Character.getNumericValue(value);
            if (getValue >= 0 && getValue <= 9) {
                pfc.click(keybordValues[getValue]);

            pfc.sleep(delayDigitInput);

float calculateNewPrice(float orderPrice) {
    if (PodFull) return priceLot - CustomFull;
    if (Barrier && orderPrice + CustomZap > BarrierPrice) return orderPrice;
    if (RoundTn) {
        boolean isWholeNumber = orderPrice == Math.floor(orderPrice);
        return (float) (Math.ceil(orderPrice * 10) / 10 + (isWholeNumber ? CustomRoundTn : 0));
    }
    if (RoundFl) {
        return Math.ceil(orderPrice) == orderPrice ? orderPrice + 1 : (float) Math.ceil(orderPrice);
    }
    if (Fifcent) {
        return orderPrice + ((priceLot - orderPrice) * (FifcentValue / 100));
    }
    return orderPrice + CustomZap;
}

    oldBalanceGold = BalanceGold;
    pfc.sleep(delayMainLoop);

    if (System.currentTimeMillis() - lastUpdateTime > delayLotCheck && !RareOrd) {
        if (Error) {
            logErrorInfo();
        }

        relogMethod();

        if (pfc.getColor(Back) == colorBack) {
            pfc.click(Back);
            pfc.sleep(100);

        } else if (pfc.getColor(Ok) == colorOk) {
            pfc.click(Ok);
            pfc.sleep(100);
        }

        ssc = true;
        lastUpdateTime = System.currentTimeMillis();

    } else if (RareOrd) {
        relogMethod();

        try {
            ActivitiesPriceFloat = Float.parseFloat(pfc.getText(LeftTopActiv, RightBotActiv));
        } catch (Exception ДастикЛучшийКодер) {}

        if (ActivitiesPriceFloat > oldActiv && oldActiv > 0) {
            ssc = true;
        } else if (ActivitiesPriceFloat < oldActiv && oldActiv > 0) {
            ssc = true;
        }

        oldActiv = ActivitiesPriceFloat;
        pfc.sleep(delayMainLoop);
    }
}
